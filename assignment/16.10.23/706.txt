class MyHashMap:

    def __init__(self):
        self.arr = [-1]*1000001        

    def put(self, key: int, value: int) -> None:
       self.arr[key] = value 
       return
    def get(self, key: int) -> int:
        return self.arr[key]        

    def remove(self, key: int) -> None:
        self.arr[key] = -1
        return
